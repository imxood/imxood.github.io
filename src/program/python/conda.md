## conda activate 指令无效

conda init powershell

## 列出当前环境中的 python 列表

conda list python

## 查看当前安装的所有 env

conda info --envs

## 安装 特定 python 版本

conda search python

conda create --name om6681 python=3.9.6
conda activate om6681
conda deactivate

conda create --name py310 python=3.10
conda activate py310
